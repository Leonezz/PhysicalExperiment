                        \includegraphics[scale=0.6]{1.png}

                        \includegraphics[scale = 0.6]{2.png}
                        \label{fig:label}
                    \end{center}

                    \begin{itemize}
                        \item 同步清零：\overline{CLR}=0+时钟上升沿
                        \item 同步置数：\overline{LD}=0+时钟上升沿
